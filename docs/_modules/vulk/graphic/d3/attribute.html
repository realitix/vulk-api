    '''Attributes is the base class for all attributes container.
    `Environment` and `Material` are attribute container
    '''

    def __init__(self, attributes=None):
        '''
        *Parameters:*

        - `attributes`: `list` of attributes ot initialize this object
        '''
        self.attributes = {}

        if attributes:
            for a in attributes:
                self.set(a)
